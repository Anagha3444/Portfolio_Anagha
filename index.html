class TransformerBlock:
    def __init__(self, dim, heads):
        self.attn = MultiHeadAttention(dim, heads)
        self.ffn = FeedForward(dim)
    
    def forward(self, x):
        x = x + self.attn(x)
        x = x + self.ffn(x)
        return x
